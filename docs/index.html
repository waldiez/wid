// install

Get Started

bash 
cargo install --git https://github.com/waldiez/wid

# Generate one ID
wid next

# Stream 10 IDs, 1s apart
wid stream --count 10

# Canonical key=value form
wid A=next W=4 Z=6 T=sec
bash 
pip install git+https://github.com/waldiez/wid

# CLI
python -m wid A=next W=4 Z=6 T=sec
python -m wid A=stream N=10 L=1

# Library
from wid import WidGen
gen = WidGen()
print(gen.next())
bash / ts 
npm install github:waldiez/wid

// ESM
import { WidGen } from 'wid';
const gen = new WidGen();
console.log(gen.next());

// CLI (after npm run build)
node typescript/dist/cli.js A=next W=4 Z=6
bash / go 
go install github.com/waldiez/wid/go/cmd/wid@latest

// As a library
import wid "github.com/waldiez/wid/go"

g := wid.NewWidGen(4, 6)
fmt.Println(g.Next())
c 
// Copy c/include/wid.h into your project — that's it.
#include "wid.h"

wid_gen_t gen;
wid_gen_init(&gen, /*W=*/4, /*Z=*/6);

char buf[64];
wid_next(&gen, buf, sizeof(buf));
printf("%s\n", buf);
bash 
# Build (uses Rust reference implementation)
docker build -t wid .

# Run
docker run --rm wid next
docker run --rm wid A=stream N=5 L=1
